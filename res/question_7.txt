func void insertionSort(int[] arr, int n) {
    int i, key, j;
    i = 1;
    for i up to n - 1 {
        if (arr[j] > key) then {
            arr[j + 1] = arr[j];  
            j = j - 1;
        } else then {
            arr[j + 1] = key;
        }
    }
} 


func void printArray(int arr[], int n) {
    int i = 0;
    for i up to n - 1 {
        print(arr[i] + " ");
    }
}
